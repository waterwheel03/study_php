var person = {
	name: 'John',
	sayHello: function(){
		console.log('Hello, ' + this.name + '!');
	}
};


//改良
var Person = function(name){
	this.name = name;
	this.sayHello = function(){
		console.log('Hello, ' + this.name + '!');
	};
};

//newのイメージ
var Person__ = function(name){
	var o = {
		name: name;
		sayHello: function(){
			console.log('Hello, ' + this.name + '!');
		}
	};
	return o;
}

var j = new Person('John');　　　：newのやつは大文字で始まる
//newのイメージ
var j2 = Person__('John');

console.log(j.name);
